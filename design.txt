1.设计数据库表,添加测试数据
    用户表[users]
	u_id       | int(4)				用户主键
	u_name     | varchar(8)			账号
	u_password | varchar(20)		密码（加密）
	u_realname | varchar(20)		真实姓名
	u_regdt    | timestamp			注册时间
	u_phone    | varchar(11)		电话
	u_role     | int(1)				角色
	u_mark     | varchar(20)		备注
	
	分类表[booktypes]
	bt_id   | int(4)				分类主键
	bt_name | varchar(50)			类型名称
	
	图书表[books]
	b_id        | int(8)			流水号
	b_isbn      | varchar(13)		编号
	b_name      | varchar(50)		书名
	b_author    | varchar(30)		作者
	b_publisher | varchar(20)		出版社
	b_price     | float				价格
	bt_id       | int(8)			书籍类型
	b_image     | varchar(50)		图片路径
	b_stock     | int(6)			库存
	b_content   | varchar(2000) 	简介
	
	   
	订单表[orders]
	o_id     	| int(4)			订单编号主键
	o_isbn      | varchar(32)		订单编号
	o_addr     	| varchar(100)		收货地址
	o_date   	| timestamp			日期
	o_mark   	| varchar(50)		备注
	u_name		| varchar(8)		订单所属用户
	o_state		| int(1)			状态
    
	
	购物车
	c_id		| int(6)			购物项id
	u_name		| varchar(8)		购物车所属用户
	b_id		| int(8)			图书id	
	b_number	| int(3)			购物项数量
	
	
    订单明细表[orderdetails]
	od_number	|int(3)				购物项数量
	o_isbn		|varchar(32)		购物项所属订单
	b_id		|int(8)				购物项所属图书
